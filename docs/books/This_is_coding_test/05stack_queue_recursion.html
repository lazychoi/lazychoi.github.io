def factorial_resursion(n):
    if n <= 1:
        print('1', end='')
        return 1
    print(n, '*', end=' ')
    return n * factorial_resursion(n-1)

print(' = ', factorial_resursion(5) ) # 5 * 4 * 3 * 2 * 1 =  120
